140            public static List<NameValuePair> getTakeOrderParams(TicketOrder order, String randcode) {
141                    List<NameValuePair> parameters = new LinkedList<NameValuePair>();
142                    parameters.add(new BasicNameValuePair("org.apache.struts.taglib.html.TOKEN", order.getToken()));
143                    parameters.add(new BasicNameValuePair("textfield", order.getTextField()));
144                    parameters.add(new BasicNameValuePair("checkbox0", order.getCheckbox0()));
145                    //è½¦æ¬¡ä¿¡æ¯
146                    parameters.add(new BasicNameValuePair("orderRequest.train_date", order.getTravelDate()));
147                    parameters.add(new BasicNameValuePair("orderRequest.train_no", order.getTrainNo()));
148                    parameters.add(new BasicNameValuePair("orderRequest.station_train_code", order.getTrainCode()));
149                    parameters.add(new BasicNameValuePair("orderRequest.from_station_telecode", order.getDepartureStationCode()));
150                    parameters.add(new BasicNameValuePair("orderRequest.to_station_telecode", order.getArrivalStationCode()));
151                    parameters.add(new BasicNameValuePair("orderRequest.seat_type_code", order.getSeatType()));
152                    parameters .add(new BasicNameValuePair("orderRequest.ticket_type_order_num", order.getTicketOrderNum()));
153                    parameters.add(new BasicNameValuePair("orderRequest.bed_level_order_num", order.getLevelOrderNum()));
154                    //å‘åˆ°ç«™ä¿¡æ?
155                    parameters.add(new BasicNameValuePair("orderRequest.start_time", order.getDepartureStationTime()));
156                    parameters.add(new BasicNameValuePair("orderRequest.end_time", order.getArrivalStationTime()));
157                    parameters.add(new BasicNameValuePair("orderRequest.from_station_name", order.getDepartureStation()));
158                    parameters.add(new BasicNameValuePair("orderRequest.to_station_name", order.getArrivalStation()));
159                    parameters.add(new BasicNameValuePair("orderRequest.cancel_flag", order.getCancelFlag()));
160                    parameters.add(new BasicNameValuePair("orderRequest.id_mode", order.getIdMode()));
161                    
162                    //è®¾ç½®ä¹˜è½¦äººä¿¡æ?
163                    List<Passenger> passengers = order.getPassengers();
164                    //ä¸?¬¡æœ?¤šåªèƒ½è´­ä¹°5å¼ è½¦ç¥?
165                    for (int i = 0; i < Config.MAXPASSENGER; i++) {
166                            Passenger passenger = (passengers.size() >= (i+1)) ? passengers.get(i) : null;
167                            if(null != passenger) {
168                                    parameters.add(new BasicNameValuePair("passengerTickets", passenger.getPassengerTickets()));
169                            }
170                            parameters.add(new BasicNameValuePair("oldPassengers", null != passenger ? passenger.getOldPassengers() : ""));
171                            if(null != passenger) {
172                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_seat", passenger.getPassengerSeat()));
173                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_ticket", passenger.getPassengerTicket()));
174                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_name", passenger.getPassengerName()));
175                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_cardtype", passenger.getPassengerCardtype()));
176                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_cardno", passenger.getPassengerCardno()));
177                                    parameters.add(new BasicNameValuePair("passenger_"+(i+1)+"_mobileno", passenger.getPassengerMobileno()));
178                            }
179                            parameters.add(new BasicNameValuePair("checkbox9", "Y"));
180                    }
181                    parameters.add(new BasicNameValuePair("randCode", randcode));
182                    parameters.add(new BasicNameValuePair("orderRequest.reserve_flag", order.getReserveFlag()));
183                    return parameters;
184            }
