478                    }

480                    String strIsLogin = HtmlUtils.getVar("isLogin", html);

482                            boolean stIsLogin = Boolean.parseBoolean(strIsLogin);

484                                    result.setCode(Result.SC_UNAUTHORIZED);

486                            }
487                    }

492                            }
493                    }
494                    return result;
495            }
